#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstdio>
#include<string>
#include<vector>
#include<set>
using namespace std;

int main(void){
	ios::sync_with_stdio(false); cin.tie(0);
	
	set<int> s;
	for(int i=0;i<3;i++){
		int n;
		cin>>n;
		s.insert(n);
	}
	int co=0;
	set<int>::iterator it;
	for(it=s.begin();it!=s.end();it++){
		co++;
	}
	cout<<co<<endl;
