#include<iostream>
using namespace std;
int main(void){
	int array[4]={1,9,7,4};
	int i,n,j,co=0;
	for(i=0;i<4;i++){
		cin>>n;
		if(n!=1&&n!=7&&n!=9&&n!=4){
			co=0;
			break;
		}
		for(j=0;j<4;j++){
			if(array[j]==n){
				array[j]=0;
//				cout<<array[0]<<array[1]<<array[2]<<array[3]<<endl;
				co++;
				break;
			}
		}
	}
	if(array[0]+array[3]+array[2]+array[1]==0)cout<<"YES"<<endl;
	else cout<<"NO"<<endl;
	
	return 0;
}
