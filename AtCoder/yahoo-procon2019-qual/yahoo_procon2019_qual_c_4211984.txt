#include<iostream>
#include<algorithm>
#include<cmath>

#include<string>
#include<vector>
#include<set>
#include<cctype>
using namespace std;

int main(void){
	ios::sync_with_stdio(false); cin.tie(0);
	
	int i,k,a,b;
	cin>>k>>a>>b;
	if(a>=b-2){
		cout<<k+1<<endl;
		return 0;
	}
	long long int bis=1,money=0;
	for(i=0;i<k;i++){
		if(money>0){
			money--;
			bis+=b;
		}
		else if(i==k-1) bis++;
		else if(bis>=a){
			bis-=a;
			money++;
		}
		else bis++;
	}
	cout<<bis<<endl;
	
	
	
	
	
	return 0;
